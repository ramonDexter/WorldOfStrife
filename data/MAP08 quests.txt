//  quest 01  //////////////////////////////////////////////////////////////////
//  sezen kuchare do hospody  //////////////////////////////////////////////////
// Find cook for Council Club
//- hospodsky zada quest hraci
//>> script: dej hraci token _given
script 2801 (void) {
    GiveActorInventory(0, "m08q_CCinnkeeper_given", 1);
    displayQuest("[ Find cook for Council Club ]", 1);
}
//v dialogu u npc kuchare
//- hrac se pta npc u jednoho bude mit moznost pod podminkou na _given
//- npc akceptuje nabidku
//>> script: dej hraci token _naselkuchare
script 2802 (void) {
    GiveActorInventory(0, "m08q_CCinnkeeper_naselkuchare", 1);
    displayQuest("[ Find cook for Council Club - return to the innkeeper. You've found a new cook. ]", 2);
}
//- hrac se crati k hospodskymu, moznost pod podminkou _naselkuchare
//>> script: dej hraci odmenu, dej hraci token _finished
script 2803 (void) {
    GiveActorInventory(0, "m08q_CCinnkeeper_finished", 1);
    GiveActorInventory(0, "m08q_duveraMistnich", 1);
    GiveActorInventory(0, "coin", 350);
    displayQuest("[ Find cook for Council Club ]", 3);
    TeleportOther(185, 186, 0); //teleport cook to kitchen
}
